Sub CYB001_GenerarDocumentos_HSBC()

Sub ReemplazarEnWord(doc As Object, target As String, replacement As String)
    With doc.Content.Find
        .Text = target
        .Replacement.Text = IIf(replacement = "", " ", replacement)
        .Execute Replace:=2 ' wdReplaceAll
    End With
End Sub

Sub InsertarImagenWord(doc As Object, bookmark As String, path As String)
    Dim rng As Object
    Set rng = doc.Content
    If rng.Find.Execute(FindText:=bookmark) Then
        rng.Text = ""
        On Error Resume Next
        doc.InlineShapes.AddPicture FileName:=path, LinkToFile:=False, SaveWithDocument:=True, Range:=rng
        On Error GoTo 0
    End If
End Sub

Function CleanFileName(str As String) As String
    Dim chars As Variant, i As Long
    ' Elimina caracteres que Windows no permite en nombres de archivos
    chars = Array("/", "\", ":", "*", "?", """", "<", ">", "|")
    For i = LBound(chars) To UBound(chars)
        str = Replace(str, chars(i), "_")
    Next i
    CleanFileName = str
End Function
